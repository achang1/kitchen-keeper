# Anonymous query of all users

query {
  allUsers {
    id
    userName
    email
    firstName
    lastName
  }
}

# Anonymous query of all users. Can query by id, user name or email

query {
  user(id:1){
    userName
    email
    firstName
    lastName
  } 
}
